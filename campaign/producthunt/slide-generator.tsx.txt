function Slide({
  id,
  children,
  dark = false,
}: {
  id: string
  children: React.ReactNode
  dark?: boolean
}) {
  return (
    <section
      id={id}
      data-slide
      className={dark ? 'dark' : ''}
      style={{
        width: 1280,
        height: 720,
        overflow: 'hidden',
        position: 'relative',
        background: dark
          ? 'radial-gradient(120% 120% at 0% 0%, #1b1b1f 0%, #0a0a0b 60%)'
          : 'radial-gradient(120% 120% at 0% 0%, #ffffff 0%, #f4f4f6 100%)',
        color: dark ? '#fafafa' : '#0a0a0b',
        marginBottom: 48,
        fontFeatureSettings: '"cv11", "ss01"',
      }}>
      <div
        aria-hidden
        style={{
          position: 'absolute',
          inset: 0,
          backgroundImage: dark
            ? 'radial-gradient(circle at 1px 1px, rgba(255,255,255,0.05) 1px, transparent 0)'
            : 'radial-gradient(circle at 1px 1px, rgba(0,0,0,0.035) 1px, transparent 0)',
          backgroundSize: '26px 26px',
          maskImage: 'radial-gradient(120% 90% at 50% 0%, black 30%, transparent 90%)',
        }}
      />
      <div style={{ position: 'relative', height: '100%', padding: 72 }}>{children}</div>
    </section>
  )
}

}

function Sub({ children, dark = false }: { children: React.ReactNode; dark?: boolean }) {

function Card({ children, className = '' }: { children: React.ReactNode; className?: string }) {
  return (
    <div
      className={`bg-background rounded-2xl border ${className}`}

/* A non-interactive prompt card matching the product's composer look. */
function ComposerCard({
  segments,
  files,
  markdown,
  footer,
}: {
  segments: Segment[]
  files?: PromptAreaFile[]
  markdown?: boolean
  footer?: 'status' | 'none'
}) {
  const [segs, setSegs] = useState<Segment[]>(segments)
  return (
    <Card>
      <div className="p-5 pb-2.5">
        <PromptArea
          value={segs}
          onChange={setSegs}
          triggers={TRIGGERS}
          files={files}
          filePosition="above"
          markdown={markdown}
          minHeight={92}
          placeholder=""
        />
        <ActionBar
          className="pt-1.5"
          left={
            <div className="text-muted-foreground flex items-center gap-2.5">
              <Plus className="size-5" />
              <AtSign className="size-[18px]" />
              <SquareSlash className="size-[18px]" />
              <Hash className="size-[18px]" />
            </div>
          }
          right={
            <div className="flex items-center gap-3">
              <span className="text-muted-foreground inline-flex items-center gap-1 text-xs font-medium">
                <Sparkles className="size-3.5" /> Opus 4.8 <ChevronDown className="size-3" />
              </span>
              <span className="bg-primary text-primary-foreground flex size-8 items-center justify-center rounded-full">
                <ArrowUp className="size-4" />
              </span>
            </div>
          }
        />
      </div>
      {footer === 'status' && (
        <StatusBar
          className="text-muted-foreground rounded-b-2xl border-t px-5 py-2.5 text-xs"
          left={
            <div className="flex items-center gap-3">
              <span className="inline-flex items-center gap-1"><GitBranch className="size-3" /> main</span>
              <span className="inline-flex items-center gap-1"><Map className="size-3" /> Plan</span>
              <span className="inline-flex items-center gap-1"><Globe className="size-3" /> WebSearch</span>
            </div>
          }
          right={<span className="inline-flex items-center gap-1"><Cloud className="size-3" /> Default</span>}
        />
      )}
    </Card>
  )
}

function Chip({ trigger, label, color }: { trigger: string; label: string; color: string }) {
  return (
    <span
      className="inline-flex items-center justify-center rounded-xl px-4 py-2 text-[32px] leading-none font-semibold"
      style={{ background: `${color}1f`, color }}>
      {trigger}
      {label}
    </span>
  )
}

/* A terminal-style command row for the dark install slide. */
function CommandRow({ children }: { children: React.ReactNode }) {
  return (
    <div
      className="flex items-center gap-3 rounded-xl border px-5 py-3.5 font-mono text-[16px]"
      style={{ background: '#141417', borderColor: '#27272a' }}>
      <span style={{ color: '#6ee7b7' }}>$</span>
      <span style={{ color: '#e4e4e7' }}>{children}</span>
    </div>
  )
}
